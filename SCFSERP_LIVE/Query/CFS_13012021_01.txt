CREATE VIEW [dbo].[VW_EXPORT_STUFF_INVOICE_PRINT_08]
AS
SELECT        dbo.TRANSACTIONMASTER.TRANMID, dbo.STUFFINGMASTER.STFBILLREFNAME
FROM            dbo.STUFFINGMASTER INNER JOIN
                         dbo.TRANSACTIONDETAIL INNER JOIN
                         dbo.TRANSACTIONMASTER ON dbo.TRANSACTIONDETAIL.TRANMID = dbo.TRANSACTIONMASTER.TRANMID INNER JOIN
                         dbo.STUFFINGDETAIL ON dbo.TRANSACTIONDETAIL.STFDID = dbo.STUFFINGDETAIL.STFDID ON 
                         dbo.STUFFINGMASTER.STFMID = dbo.STUFFINGDETAIL.STFMID
WHERE        (dbo.TRANSACTIONMASTER.SDPTID = 2)
GROUP BY dbo.TRANSACTIONMASTER.TRANMID, dbo.STUFFINGMASTER.STFBILLREFNAME


GO
